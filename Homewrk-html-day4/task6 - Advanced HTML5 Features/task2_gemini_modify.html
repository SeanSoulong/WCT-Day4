Offline Web App

This webpage is cached using a service worker, so you can access it even when offline.

----javacript----- self.addEventListener('install', event => { event.waitUntil( caches.open('static-cache') .then(cache => { return cache.addAll([ '/', 'service-worker.js', 'style.css' // Add any other static assets you want to cache ]); }) ); }); self.addEventListener('fetch', event => { event.respondWith( caches.match(event.request) .then(response => { if (response) { return response; } // Fallback to network request if not cached return fetch(event.request); }) ); });
