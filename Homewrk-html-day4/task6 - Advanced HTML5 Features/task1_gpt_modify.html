Welcome to My Offline Webpage

This webpage demonstrates the use of a service worker to cache content for offline access. You can visit this page even when you are not connected to the internet!

-------styles.css------- body { font-family: Arial, sans-serif; text-align: center; padding: 20px; background-color: #f4f4f4; } header { background-color: #4CAF50; color: white; padding: 10px; border-radius: 5px; } main { margin-top: 20px; font-size: 1.2em; } -------manifest.json------ { "name": "Offline Webpage Example", "short_name": "OfflinePage", "start_url": "/", "display": "standalone", "background_color": "#ffffff", "theme_color": "#4CAF50", "description": "A simple webpage demonstrating offline access using a service worker." } --------service-worker.js-------- const CACHE_NAME = 'v1'; const CACHE_ASSETS = [ '/', '/index.html', '/styles.css', ]; // Install event self.addEventListener('install', (event) => { event.waitUntil( caches.open(CACHE_NAME) .then((cache) => { console.log('Caching assets'); return cache.addAll(CACHE_ASSETS); }) ); }); // Fetch event self.addEventListener('fetch', (event) => { event.respondWith( caches.match(event.request) .then((response) => { return response || fetch(event.request); }) ); }); // Activate event self.addEventListener('activate', (event) => { const cacheWhitelist = [CACHE_NAME]; event.waitUntil( caches.keys().then((cacheNames) => { return Promise.all( cacheNames.map((cacheName) => { if (cacheWhitelist.indexOf(cacheName) === -1) { return caches.delete(cacheName); } }) ); }) ); });
